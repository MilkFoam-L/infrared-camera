import json
import os
import sys

import requests


thingsboard_host = os.getenv("THINGSBOARD_HOST", "192.168.1.78:8080")
device_token = os.getenv("THINGSBOARD_TOKEN")

if len(sys.argv) > 1:
    device_token = sys.argv[1]

if not device_token:
    raise SystemExit("请先设置环境变量 THINGSBOARD_TOKEN，或执行：python thingsboard上报.txt <设备访问令牌>")

thingsboard_url = f"http://{thingsboard_host}/api/v1/{device_token}/telemetry"
upload_data = {
    "warning_flag": "1",
    "warning_status": "1"
}
json_data = json.dumps(upload_data, ensure_ascii=False)
headers = {
    "Content-Type": "application/json; charset=utf-8"
}

print(f"ThingsBoard URL: {thingsboard_url}")
print(f"Payload: {json_data}")

response = requests.post(
    thingsboard_url,
    data=json_data.encode("utf8"),
    headers=headers,
    timeout=5
)

print(f"Status: {response.status_code}")
print(f"Response: {response.text}")
response.raise_for_status()

import json

import requests


thingsboard_host = "192.168.1.78:8080"
device_token = "Re7fUx8Zrr6tl69BARLZ"

if not device_token or device_token == "PLEASE_SET_THINGSBOARD_TOKEN":
    raise SystemExit("请先打开 thingsboard上报.txt，把 device_token 改成 ThingsBoard 设备访问令牌")

thingsboard_url = f"http://{thingsboard_host}/api/v1/{device_token}/telemetry"
upload_data = {
    "warning_flag": "1",
    "warning_status": "1"
}
json_data = json.dumps(upload_data, ensure_ascii=False)
headers = {
    "Content-Type": "application/json; charset=utf-8"
}

print(f"ThingsBoard URL: {thingsboard_url}")
print(f"Payload: {json_data}")

response = requests.post(
    thingsboard_url,
    data=json_data.encode("utf8"),
    headers=headers,
    timeout=5
)

print(f"Status: {response.status_code}")
print(f"Response: {response.text}")
response.raise_for_status()
